% Create fake small caps command that preserves original casing
% Expand the argument first so macro inputs (e.g. \spelltitle) are handled
\newcommand\fauxsc[1]{%
  \begingroup
  \edef\fauxsctemp{\noexpand\fauxschelper#1 \relax\relax}%
  \fauxsctemp
  \endgroup
}
\def\fauxschelper#1 #2\relax{%
  \fauxschelphelp#1\relax\relax%
  \if\relax#2\relax\else\ \fauxschelper#2\relax\fi%
}
\def\Hscale{.85}\def\Vscale{.75}\def\Cscale{1.00}
\def\fauxschelphelp#1#2\relax{%
  \ifnum`#1=\lccode`#1\relax\scalebox{\Hscale}[\Vscale]{\char\uccode`#1}\else%
    \scalebox{\Cscale}[1]{#1}\fi%
  \ifx\relax#2\relax\else\fauxschelphelp#2\relax\fi}

% Command to display range with area effect icons if applicable
\newcommand{\displayrange}[1]{%
    \ifx\relax#1\relax%
        \relax%
    \else%
        % Split the input: rangetext|areaeffect|closingparen
        \StrBefore{#1}{|}[\rangetext]%
        \StrBehind{#1}{|}[\remaining]%
        \StrBefore{\remaining}{|}[\areaeffect]%
        \StrBehind{\remaining}{|}[\closingparen]%
        % Check if there's an area effect (not "none")
        \ifnum\pdfstrcmp{\areaeffect}{none}=0%
            % No area effect icon
            {\italicfont\rangetext}%
        \else%
            % Show area effect icon and add closing parenthesis
            {\italicfont\rangetext}\hspace{0.5mm}\includesvg[width=0.15cm, keepaspectratio]{../images/areas_effect/\areaeffect}%
            \IfStrEq{\closingparen}{}{}{{\italicfont\closingparen}}%
        \fi%
    \fi%
}

% Helper command to display a single damage type icon
\newcommand{\displaydamagetypeicon}[1]{%
    \ifx\relax#1\relax%
    \else%
        \IfStrEq{#1}{}{}{\includesvg[width=0.15cm, height=0.15cm, keepaspectratio]{../images/damage_types/#1}\hspace{0.5mm}}%
    \fi%
}

% Command to display damage/effect with damage type icons
\newcommand{\displaydamageeffect}[1]{%
    \ifx\relax#1\relax%
        \relax%
    \else%
        \StrBefore{#1}{|}[\damagetext]%
        \StrBehind{#1}{|}[\damagetypes]%
        % Check if there are damage types
        \IfStrEq{\damagetypes}{}{%
            % No damage types, just display text
            {\italicfont\damagetext}%
        }{%
            % Display icons for each damage type, then the text
            \IfSubStr{\damagetypes}{acid}{\displaydamagetypeicon{acid}}{}%
            \IfSubStr{\damagetypes}{bludgeoning}{\displaydamagetypeicon{bludgeoning}}{}%
            \IfSubStr{\damagetypes}{cold}{\displaydamagetypeicon{cold}}{}%
            \IfSubStr{\damagetypes}{fire}{\displaydamagetypeicon{fire}}{}%
            \IfSubStr{\damagetypes}{force}{\displaydamagetypeicon{force}}{}%
            \IfSubStr{\damagetypes}{lightning}{\displaydamagetypeicon{lightning}}{}%
            \IfSubStr{\damagetypes}{necrotic}{\displaydamagetypeicon{necrotic}}{}%
            \IfSubStr{\damagetypes}{piercing}{\displaydamagetypeicon{piercing}}{}%
            \IfSubStr{\damagetypes}{poison}{\displaydamagetypeicon{poison}}{}%
            \IfSubStr{\damagetypes}{psychic}{\displaydamagetypeicon{psychic}}{}%
            \IfSubStr{\damagetypes}{radiant}{\displaydamagetypeicon{radiant}}{}%
            \IfSubStr{\damagetypes}{slashing}{\displaydamagetypeicon{slashing}}{}%
            \IfSubStr{\damagetypes}{thunder}{\displaydamagetypeicon{thunder}}{}%
            {\italicfont\damagetext}%
        }%
    \fi%
}
