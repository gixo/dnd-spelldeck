% Command to display range with cube icon if applicable
\newcommand{\displayrange}[1]{%
    \ifx\relax#1\relax%
        \relax%
    \else%
        \StrBefore{#1}{|}[\rangetext]%
        \StrBehind{#1}{|}[\showicon]%
        {\italicfont\rangetext}%
        \ifx\showicon\empty%
        \else%
            \ifnum\pdfstrcmp{\showicon}{True}=0%
                \hspace{1mm}\includesvg[width=0.15cm]{../images/cube}%
            \fi%
        \fi%
    \fi%
}
